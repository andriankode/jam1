srcdoc='
<html>
  <body style="margin:0; padding:0; display:flex; flex-direction:column; justify-content:center; align-items:center; height:100%; background:#fff; font-family:Arial, sans-serif; color:#202124;">
    <div id="time" style="font-size:26px; font-weight:bold;">--:--:--</div>
    <div id="date" style="font-size:12px; color:#5f6368;"></div>
    <script>
      function updateClock() {
        var now = new Date();
        var h = now.getHours();
        var m = String(now.getMinutes()).padStart(2, "0");
        var s = String(now.getSeconds()).padStart(2, "0");

        var days = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday"];
        var months = ["January","February","March","April","May","June","July","August","September","October","November","December"];

        document.getElementById("time").textContent = h + ":" + m + ":" + s;
        document.getElementById("date").textContent = days[now.getDay()] + ", " + months[now.getMonth()] + " " + now.getDate() + ", " + now.getFullYear();
      }
      setInterval(updateClock, 1000);
      updateClock();
    </script>
  </body>
</html>
